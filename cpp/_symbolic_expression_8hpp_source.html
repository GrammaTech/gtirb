168  void moveToNextBit() {
169  if (CurrIndex <= static_cast<size_t>(SymAttribute::Max)) {
170  do {
171  ++CurrIndex;
172  } while (CurrIndex <= static_cast<size_t>(SymAttribute::Max) &&
173  !SASet.Flags[CurrIndex]);
174  }
175  }
176 
177  void moveToPreviousBit() {
178  size_t NewIndex = CurrIndex;
179  while (NewIndex > 0) {
180  --NewIndex;
181  if (SASet.Flags[NewIndex])
182  break;
183  }
184  // Assert if there is no earlier bit.
185  // This would indicate an attempt to move before begin().
186  assert(SASet.Flags[NewIndex]);
187  if (SASet.Flags[NewIndex]) {
188  CurrIndex = NewIndex;
189  }
190  }
