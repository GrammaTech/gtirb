44  // SFINAE overload handling a Callable with a void return type.
45  template <typename Callable, typename Ty, typename CommonRetTy>
46  auto visit_impl_help(Callable&& Visitor) const
47  -> std::enable_if_t<std::is_void_v<CommonRetTy>, bool> {
48  static_assert(std::is_invocable_v<Callable, Ty*>,
49  "Visitor must contain an overloaded function call operator "
50  "for each of the types in supported_types");

52  if (Ty* Obj = dyn_cast_or_null<Ty>(*Ptr)) {
53  std::invoke(Visitor, Obj);
54  return true;
55  }
56  }
57  return false;
58  }

78  // Helper type traits used to determine whether all of the Callable object's
79  // return types agree. Note that the return types do not need to be the same,
80  // but do need to be implicitly convertible to the same common type. e.g.,
81  // this class is a valid Callable.
82  // struct Visitor {
83  // int operator()(Block*);
84  // long operator()(DataBlock*);
85  // };

110  // Helper function to unpack all of the types in the type list and attempt to
111  // visit Callable once per type. Verifies that the Callable objects all share
112  // a compatible return type.
113  //
114  // SFINAE overload handling a Callable where there is a common, non-void
115  // return type. Returns a std::optional<common_type>.
116  template <typename Callable, template <typename...> typename TypeList,
117  typename... Types>
118  auto visit_impl(Callable&& Visitor, TypeList<Types...>) const
119  -> std::enable_if_t<
120  !std::is_void_v<common_return_type_t<Callable, TypeList, Types...>>,
121  std::optional<common_return_type_t<Callable, TypeList, Types...>>> {
122  // If this assertion fails, the return values from the Callable object are
123  // not compatible enough. This can happen if they return incompatible types
124  // or if there is not an overload for each referent type.
125  static_assert(common_return_type_v<Callable, TypeList, Types...>,
126  "incompatible return types for the Callable object");
127  // Instantiate a call to the Visitor once for each of the listed types, but
128  // only issue the call at runtime if the Referent can be dynamically cast to
129  // the given type. In this way, the Visitor needs to be able to handle any
130  // of the supported types, but will only be called once for the concrete
131  // type of the Referent.
132  //
133  // If you get an error about there being no matching overloaded function for
134  // the call to visit_impl_help, that is most likely because there are one
135  // or more overloads missing for each referent type.
136  std::optional<common_return_type_t<Callable, TypeList, Types...>> Res;
137  (... ||
138  visit_impl_help<Callable, Types, typename decltype(Res)::value_type>(
139  std::forward<Callable>(Visitor), Res));
140  return Res;
141  }

143  // SFINAE overload handling a Callable where every return type is void.
144  // Returns void.
145  template <typename Callable, template <typename...> typename TypeList,
146  typename... Types>
147  auto visit_impl(Callable&& Visitor, TypeList<Types...>) const
148  -> std::enable_if_t<
149  std::is_void_v<common_return_type_t<Callable, TypeList, Types...>>,
150  void> {
151  // Call each of the overloads on Callable, but there is no value to be
152  // returned from any of the calls.
153  (... ||
154  visit_impl_help<Callable, Types, void>(std::forward<Callable>(Visitor)));
155  }
