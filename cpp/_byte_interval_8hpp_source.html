1563  template <typename ByteIntervalType, typename T> class BytesReference {
1564  public:
1565  BytesReference(ByteIntervalType* BI_, size_t I_,
1566  boost::endian::order InputOrder_,
1567  boost::endian::order OutputOrder_)
1568  : BI(BI_), I(I_), InputOrder(InputOrder_), OutputOrder(OutputOrder_) {}
1569 
1575  operator T() const {
1576  assert(I + sizeof(T) <= BI->Size &&
1577  "read into interval's bytes out of bounds!");
1578 
1579  auto S = BI->Bytes.size();
1580 
1581  if (I >= S) {
1582  // anything this far past the end of initialized bytes is composed of
1583  // all zero bytes, so we return what T would have been interpreted as
1584  // if all bytes are zero.
1585  //
1586  // (note that you may be tempted to replace this with "return T{};",
1587  // but beware: T might be a non-scalar type whose default constructor
1588  // differs from the value returned when all bytes are re-interpeted as
1589  // zeroes. A similar argument exists for "return T{0};".)
1590  const std::array<uint8_t, sizeof(T)> Array{};
1591  return *reinterpret_cast<const T*>(Array.data());
1592  }
1593 
1594  if (I + sizeof(T) > S) {
1595  // Here, I < S < I + sizeof(T), so we need to partially fill the
1596  // initialized bytes and combine it with zeroes for the uninitialized
1597  // bytes.
1598  std::array<uint8_t, sizeof(T)> Array{};
1599  // Thanks to math, 0 < S - I < sizeof(T).
1600  std::copy_n(BI->Bytes.begin() + I, S - I, Array.begin());
1601  return endian_flip(*reinterpret_cast<const T*>(Array.data()),
1602  InputOrder, OutputOrder);
1603  }
1604 
1605  return endian_flip(*reinterpret_cast<const T*>(BI->Bytes.data() + I),
1606  InputOrder, OutputOrder);
1607  }
1608 
1615  BytesReference<ByteIntervalType, T>& operator=(const T& rhs) {
1616  assert(I + sizeof(T) <= BI->Size &&
1617  "write into interval's bytes out of bounds!");
1618 
1619  if (I + sizeof(T) > BI->Bytes.size()) {
1620  BI->Bytes.resize(I + sizeof(T));
1621  }
1622 
1623  *reinterpret_cast<T*>(BI->Bytes.data() + I) =
1624  endian_flip(rhs, OutputOrder, InputOrder);
1625  return *this;
1626  }
1627

1629  size_t I;
1630  boost::endian::order InputOrder;
1631  boost::endian::order OutputOrder;
1632  };
