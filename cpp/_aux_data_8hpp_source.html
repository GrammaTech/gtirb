149 class FromByteRange {
150 public:
151  explicit FromByteRange(const std::string& Bytes)
152  : Curr(Bytes.begin()), End(Bytes.end()) {}
153 
154  bool read(std::byte& Byte) {
155  if (Curr == End)
156  return false;
157 
158  Byte = std::byte(*Curr);
159  ++Curr;
160  return true;
161  }
162 
163  uint64_t remainingBytesToRead() const {
164  return static_cast<uint64_t>(std::distance(Curr, End));
165  }
166 
167 private:
168  std::string::const_iterator Curr;
169  std::string::const_iterator End;

215 // Serialize and deserialize by copying the object representation directly.
216 template <typename T>
217 struct default_serialization<
218  T, typename std::enable_if_t<is_endian_type<T>::value>> {
219  static void toBytes(const T& object, ToByteRange& TBR) {
220  // Store as little-endian.
221  T reversed = boost::endian::conditional_reverse<
222  boost::endian::order::little, boost::endian::order::native>(object);
223  auto srcBytes_begin = reinterpret_cast<std::byte*>(&reversed);
224  auto srcBytes_end = reinterpret_cast<std::byte*>(&reversed + 1);
225  std::for_each(srcBytes_begin, srcBytes_end, [&](auto b) { TBR.write(b); });
226  }
227 
228  static bool fromBytes(T& object, FromByteRange& FBR) {
229  auto dest_begin = reinterpret_cast<std::byte*>(&object);
230  auto dest_end = reinterpret_cast<std::byte*>(&object + 1);
231  bool Success = true;
232  std::for_each(dest_begin, dest_end, [&](auto& b) {
233  if (!FBR.read(b))
234  Success = false;
235  });
236  if (!Success) {
237  return false;
238  }
239 
240  // Data stored as little-endian.
241  boost::endian::conditional_reverse_inplace<boost::endian::order::little,
242  boost::endian::order::native>(
243  object);
244  return true;
245  }
246 };

590  static std::unique_ptr<AuxData> fromProtobuf(const MessageType& Message) {
591  // Check if the serialized type isn't compatible with the type
592  // we're trying to deserialize to.
593  if (!checkAuxDataMessageType(
594  Message, auxdata_traits<typename Schema::Type>::type_name())) {
595  return nullptr;
596  }
597 
598  // Note: Do not access Message's contents here. That would introduce
599  // dllexport/dllimport problems on Windows. Call the base class's
600  // fromProtobuf function and then unserialize from its
601  // SerializedForm structure.
602  auto TypedAuxData = std::make_unique<AuxDataImpl<Schema>>();
603  AuxData::fromProtobuf(*TypedAuxData, Message);
604  FromByteRange FBR(TypedAuxData->rawData().RawBytes);
605  if (!auxdata_traits<typename Schema::Type>::fromBytes(TypedAuxData->Object,
606  FBR))
607  return nullptr;
608  return TypedAuxData;
609  }

614  virtual void toProtobuf(MessageType* Message) const override {
615  // Note: Do not edit Message's contents here. That would introduce
616  // dllexport/dllimport problems on Windows. Store to a
617  // SerializedForm, and then call the base class's toProtobuf
618  // function.

620  TypedSF.ProtobufType = auxdata_traits<typename Schema::Type>::type_name();

622  auxdata_traits<typename Schema::Type>::toBytes(this->Object, TBR);
