71 // Helper for constructing the CFG type. The graph property needs to refer to
72 // the graph's vertex_descriptor type. This is accessible via
73 // boost::adjacency_list_traits, but requires keeping the template parameters
74 // for boost::adjacency_list and boost::adjacency_list_traits in sync. This
75 // helper ensures the relevant parameters are the same for both.
76 template <class OutEdgeListS = boost::vecS, class VertexListS = boost::vecS,
77  class DirectedS = boost::directedS, class EdgeListS = boost::listS>
78 struct CfgBuilder {
79  using vertex_descriptor = typename boost::adjacency_list_traits<
80  OutEdgeListS, VertexListS, DirectedS, EdgeListS>::vertex_descriptor;
81  using type = boost::adjacency_list<
82  OutEdgeListS, VertexListS, DirectedS,
83  // Vertices are CfgNodes.
84  CfgNode*,
85  // Edges have labels.
86  EdgeLabel,
87  // The graph keeps track of vertex descriptors for
88  // each node.
89  std::unordered_map<const CfgNode*, vertex_descriptor>, EdgeListS>;
90 };

317 template <class Traits, typename CfgNodePtr> struct EdgeDescrToNodeLabel {
318  // Yes, this stores a const CFG* even for the version that returns a
319  // non-const CfgNode. We can do this because the constness of the two are
320  // actually independent, though we maintain the illusion of their related
321  // constness in the public interface functions cfgPredecessors and
322  // cfgSuccessors.
323  const CFG* Cfg = nullptr; // Should only be null for end iterator
324 
325  EdgeDescrToNodeLabel() = default;
326  EdgeDescrToNodeLabel(const CFG* G) : Cfg(G) {}
327  // This quasi-copy constructor enables conversion from non-const iterator to
328  // const iterator, but not vice versa.
329  EdgeDescrToNodeLabel(const EdgeDescrToNodeLabel<Traits, CfgNode*>& Rhs)
330  : Cfg(Rhs.Cfg) {}
331 
332  std::pair<CfgNodePtr, EdgeLabel>
333  operator()(const CFG::edge_descriptor& EDesc) const {
334  return {Traits::getNode(EDesc, *Cfg), (*Cfg)[EDesc]};
335  }

353 template <class Traits, typename CfgNodePtr>
354 auto cfgEdgeIters(const CFG& G, const CfgNode* N) {
355  const std::optional<CFG::vertex_descriptor> OptVtxDescr = getVertex(N, G);
356  if (OptVtxDescr == std::nullopt) {
357  // FYI: this is the return type
358  return boost::iterator_range<
359  boost::transform_iterator<EdgeDescrToNodeLabel<Traits, CfgNodePtr>,
360  typename Traits::edge_iterator>>();
361  } else {
362  const auto [Begin, End] = Traits::getEdges(OptVtxDescr.value(), G);
363  return boost::make_iterator_range(
364  boost::make_transform_iterator(
365  Begin, EdgeDescrToNodeLabel<Traits, CfgNodePtr>{&G}),
366  boost::make_transform_iterator(
367  End, EdgeDescrToNodeLabel<Traits, CfgNodePtr>{&G}));
368  }
369 }
